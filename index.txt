        // ==========================================
        // SCROLL REVEAL ANIMATION
        // ==========================================
        function revealOnScroll() {
            const elements = document.querySelectorAll('.connect-card, .feed-container');
            
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.classList.add('revealed');
                    }
                });
            }, {
                threshold: 0.1,
                rootMargin: '0px 0px -50px 0px'
            });
            
            elements.forEach(el => observer.observe(el));
        }

        // ==========================================
        // ELFSIGHT LOAD MORE BUTTON - VISIBILITY FIX
        // ==========================================
        function fixLoadMoreButton() {
            const widget = document.querySelector('.elfsight-app-ff4f7668-f218-4440-b095-f94a3e38fba3');
            if (!widget) return;
            
            const allButtons = widget.querySelectorAll('button');
            
            allButtons.forEach(function(button) {
                const text = button.textContent.toLowerCase().trim();
                
                const isLoadMore = text.includes('load more') || 
                                   text.includes('loadmore') || 
                                   text.includes('show more') || 
                                   text.includes('view more') ||
                                   text.includes('see more') ||
                                   text === 'more';
                
                if (isLoadMore || button.className.includes('load-more') || button.className.includes('loadmore')) {
                    
                    button.style.setProperty('background', '#1A1A1A', 'important');
                    button.style.setProperty('background-color', '#1A1A1A', 'important');
                    button.style.setProperty('color', '#FFFFFF', 'important');
                    button.style.setProperty('border', '1px solid #EAE5DC', 'important');
                    button.style.setProperty('border-radius', '60px', 'important');
                    button.style.setProperty('padding', '14px 36px', 'important');
                    button.style.setProperty('font-family', "'Inter', sans-serif", 'important');
                    button.style.setProperty('font-size', '13px', 'important');
                    button.style.setProperty('font-weight', '500', 'important');
                    button.style.setProperty('letter-spacing', '0.5px', 'important');
                    button.style.setProperty('cursor', 'pointer', 'important');
                    button.style.setProperty('opacity', '1', 'important');
                    button.style.setProperty('visibility', 'visible', 'important');
                    button.style.setProperty('display', 'inline-block', 'important');
                    button.style.setProperty('transition', 'all 0.3s ease', 'important');
                    
                    button.addEventListener('mouseenter', function() {
                        this.style.setProperty('background', '#333333', 'important');
                        this.style.setProperty('border-color', '#C9A96E', 'important');
                        this.style.setProperty('transform', 'translateY(-2px)', 'important');
                        this.style.setProperty('box-shadow', '0 4px 16px rgba(201, 169, 110, 0.2)', 'important');
                    });
                    
                    button.addEventListener('mouseleave', function() {
                        this.style.setProperty('background', '#1A1A1A', 'important');
                        this.style.setProperty('border-color', '#EAE5DC', 'important');
                        this.style.setProperty('transform', 'translateY(0)', 'important');
                        this.style.setProperty('box-shadow', 'none', 'important');
                    });
                }
            });
        }

        // ==========================================
        // OBSERVER - Watch for widget changes
        // ==========================================
        function setupObserver() {
            const widget = document.querySelector('.elfsight-app-ff4f7668-f218-4440-b095-f94a3e38fba3');
            if (!widget) {
                setTimeout(setupObserver, 1000);
                return;
            }
            
            const observer = new MutationObserver(function() {
                fixLoadMoreButton();
            });
            
            observer.observe(widget, { 
                childList: true, 
                subtree: true,
                attributes: false
            });
        }

        // ==========================================
        // INIT
        // ==========================================
        document.addEventListener('DOMContentLoaded', function() {
            setTimeout(fixLoadMoreButton, 1500);
            setTimeout(fixLoadMoreButton, 3000);
            setTimeout(fixLoadMoreButton, 5000);
            setTimeout(setupObserver, 2000);
            revealOnScroll();
            
            let scrollTimeout;
            window.addEventListener('scroll', function() {
                clearTimeout(scrollTimeout);
                scrollTimeout = setTimeout(fixLoadMoreButton, 300);
            });
        });
    </script>
